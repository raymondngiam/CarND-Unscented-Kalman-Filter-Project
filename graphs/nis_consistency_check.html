def calc_rmse(data1,data2):
    rmse=data1-data2
    rmse=rmse*rmse
    rmse=np.sqrt(rmse.mean())
    return rmse

rmse_x=calc_rmse(log_predictions["x"],log_predictions["x_gt"])

def plot_state_ground_truth(log,name):
    plt.xlabel("time step, k")
    plt.ylabel(name)
    plt.grid()
    plt.plot(log[name], '-', color="r",label=name)
    plt.plot(log[name+"_gt"], '-', color="b",label=name+" ground truth")
    plt.legend(loc="best") 
    plt.title(name)

fig = plt.figure(figsize=(14, 12))
fig.add_subplot(2,2,1)
plot_state_ground_truth(log_predictions,"x")
fig.add_subplot(2,2,2)
plot_state_ground_truth(log_predictions,"y")
fig.add_subplot(2,2,3)
plot_state_ground_truth(log_predictions,"vx")
fig.add_subplot(2,2,4)
plot_state_ground_truth(log_predictions,"vy")
